def to_sparql(**options)
  str = to_triple.map {|term| term.to_sparql(**options)}.join(" ")
  quoted? ? ('<<' + str + '>>') : str
end

def to_sxp_bin
  [ (has_graph? ? :quad : (quoted? ? :qtriple : :triple)),
    (:inferred if inferred?),
    subject,
    predicate,
    object,
    graph_name
  ].compact.map(&:to_sxp_bin)
end
